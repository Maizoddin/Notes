In Function constructor if we add any property directly inside that function using this then it will be present inside that. but for each object of that it will create new one(waste of memory).

function Fun(name){
    this.name = name
    this.specific = function (){
        console.log("SSS")
    }
}

const f = new Fun("Maaz")

// console f
{name:"Maaz", specific:f}

// console f.__proto__
{}

Fun.prototype.greet = function(){
    console.log(this.name)
}

//  console f.__proto__
{greet:f}
